The Integrated Schema

-- EVENT LAYER
event_log (
    event_id, transaction_id,
    event_type,        -- Links to capability_clusters
    actor_id, role,
    event_hash, previous_hash  -- Immutable chain
)

-- POLICY LAYER
policies (
    policy_id, rule_name, threshold,
    required_role,     -- Links to functional_roles
    sla_hours
)

policy_result (
    result_id, event_id, policy_id,
    result: PASS | FAIL | BYPASS
)

-- LEDGER LAYER
transactions (
    id, amount, vendor,
    account_id,         -- Links to chart_of_accounts
    status, current_stage,
    transaction_hash    -- Latest hash for integrity
)

-- CAPACITY LAYER (NEW)
capability_clusters (
    cluster_id, cluster_code,
    event_type,        -- Links to event_log.event_type
    base_effort_min,
    required_skill_level,
    automation_eligible
)

staff_profiles (
    staff_id, name,
    functional_role,   -- Links to policies.required_role
    skill_levels,       -- Matrix by domain
    weekly_capacity_hrs
)

-- THE CONNECTIONS
event_log.event_type → capability_clusters.event_type
policies.required_role → staff_profiles.functional_role
capability_clusters.base_effort_min × volume = capacity_required
staff_profiles.weekly_capacity_hrs × headcount = capacity_available
